const multiplydigits = num => {
  let newnum=1;
  while(num > 0) {
    let digit = num % 10;
    newnum *= digit;
    num = (num - digit)/10;
  }
  return newnum;
  };

const persistence = num => {
  let result = Infinity, steps = 0;
  if (num < 10) {
    return 0;
  }
  while(result >= 10) {
    result = multiplydigits(num);
    num = result;
    steps += 1;
  }
  return pers;
}
